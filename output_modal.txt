          <!-- 업데이트 타입 -->

  <script>
    
    let selectedFile = null;
    let fileChecksum = null;
    let isVerified = false;

    function handleFileSelect(event) {
      const file = event.target.files[0];
      if (file) {
        selectedFile = file;
        document.getElementById('fileDisplay').textContent = file.name;
        document.getElementById('fileDisplay').classList.add('has-file');
        document.getElementById('fileSizeDisplay').textContent = formatFileSize(file.size);
        document.getElementById('fileClear').style.display = 'block';
        document.getElementById('verifyButton').disabled = false;
        
        // Reset verification
        isVerified = false;
        document.getElementById('checksumInfo').style.display = 'none';
        updateRegisterButton();

        // Auto-calculate checksum
        calculateChecksum(file);
      }
    }

    function clearFile() {
      selectedFile = null;
      fileChecksum = null;
      isVerified = false;
      document.getElementById('artifactFile').value = '';
      document.getElementById('fileDisplay').textContent = '파일을 선택해 주세요.';
      document.getElementById('fileDisplay').classList.remove('has-file');
      document.getElementById('fileSizeDisplay').textContent = '0';
      document.getElementById('fileClear').style.display = 'none';
      document.getElementById('verifyButton').disabled = true;
      document.getElementById('checksumInfo').style.display = 'none';
      updateRegisterButton();
    }

    async function calculateChecksum(file) {
      try {
        const arrayBuffer = await file.arrayBuffer();
        const hashBuffer = await crypto.subtle.digest('SHA-256', arrayBuffer);
        const hashArray = Array.from(new Uint8Array(hashBuffer));
        const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
        fileChecksum = hashHex;
        document.getElementById('checksumValue').textContent = hashHex;
        document.getElementById('checksumInfo').style.display = 'block';
      } catch (error) {
        console.error('Checksum calculation failed:', error);
        alert('체크섬 계산 중 오류가 발생했습니다.');
      }
    }

    function verifyFile() {
      if (selectedFile && fileChecksum) {
        isVerified = true;
        alert('파일 확인이 완료되었습니다.\n\n' +
              '파일명: ' + selectedFile.name + '\n' +
              '크기: ' + formatFileSize(selectedFile.size) + '\n' +
              '체크섬: ' + fileChecksum.substring(0, 16) + '...');
        updateRegisterButton();
      }
    }

    function formatFileSize(bytes) {
      if (bytes === 0) return '0 Bytes';
      const k = 1024;
      const sizes = ['Bytes', 'KB', 'MB', 'GB'];
      const i = Math.floor(Math.log(bytes) / Math.log(k));
      return Math.round(bytes / Math.pow(k, i) * 100) / 100 + ' ' + sizes[i];
    }

    function updateRegisterButton() {
      const button = document.getElementById('registerButton');
      button.disabled = !isVerified;
    }

    function registerVersion() {
      const major = document.getElementById('versionMajor').value;
      const minor = document.getElementById('versionMinor').value;
      const patch = document.getElementById('versionPatch').value;
      const updateType = document.querySelector('input[name="updateType"]:checked').value;
      const deploymentTarget = document.querySelector('input[name="deploymentTarget"]:checked').value;
      const versionState = document.querySelector('input[name="versionState"]:checked').value;
      const releaseNotes = document.getElementById('releaseNotes').value;

      if (!selectedFile || !isVerified) {
        alert('파일을 선택하고 확인해 주세요.');
        return;
      }

      const formData = {
        version_major: parseInt(major),
        version_minor: parseInt(minor),
        version_patch: parseInt(patch),
        version_tag: `${major}.${minor}.${patch}`,
        update_type: updateType,
        deployment_target: deploymentTarget,
        checksum: fileChecksum,
        artifact_filename: selectedFile.name,
        artifact_size_bytes: selectedFile.size,
        release_notes: releaseNotes,
        state: versionState
      };

      console.log('Registering version:', formData);
      
      // TODO: Actual API call
      // await fetch('/api/contents/${contentId}/versions', {
      //   method: 'POST',
      //   body: formData with file
      // });

      alert('버전이 등록되었습니다.\n\n버전: ' + formData.version_tag + 
            '\n업데이트: ' + (updateType === 'forced' ? '강제' : '선택') +
            '\n배포: ' + (deploymentTarget === 'production' ? '상용' : '테스트'));
      
      closeModal();
    }

    function closeModal() {
      // In a real app, this would close the modal or navigate back
      if (confirm('작성 중인 내용이 사라집니다. 닫으시겠습니까?')) {
        window.location.href = 'content-editor.html';
      }
    }
  
    
    function openModal() {
       document.getElementById('versionModal').style.display = 'flex';
    }
    function closeModal() {
       document.getElementById('versionModal').style.display = 'none';
    }

  </script>
